# 9 [Palindrome Number]

def leet9(x):
    if x<0:
        return False
    sub=x[::-1]
    if x==sub:
        return True
    return False

# 455 [Assign Cookies]

<링크>open in new window

def leet455(g,s):
    g=sorted(g)
    s=sorted(s)
    idx=0
    for cookie in s:
        if g[idx]>cookie:
            continue
        else:
            idx+=1
            if idx>=len(g):
                break
    return idx

# 11 [Container With Most Water]

<링크>open in new window

def leet12(height):
    leftX=0
    rightX=len(height)-1
    leftMax=height[leftX]
    rightMax=height[rightX]
    maxWater=0
    for i in range(len(height)):
        if leftMax<=height[i] and height[leftX]<=rightMax:
            nowMaxWater=min(height[i],rightMax)*(rightX-i)
            if maxWater<=nowMaxWater:
                leftX=i
                leftMax=height[leftX]
                maxWater=nowMaxWater
        if rightMax<=height[len(height)-i-1] and height[rightX]<=leftMax:
            nowMaxWater=min(leftMax,height[len(height)-i-1])*(len(height)-i-1-leftX)
            if maxWater<=nowMaxWater:
                rightX=len(height)-i-1
                rightMax=height[rightX]
                maxWater=nowMaxWater
        if leftX>=rightX:
            break
    return maxWater
