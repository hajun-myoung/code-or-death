def solution(s):
    answer = ""
    words = {
        "zero" : "0",
        "one": "1",
        "two": "2",
        "three": "3",
        "four": "4",
        "five": "5",
        "six": "6",
        "seven": "7",
        "eight": "8",
        "nine": "9",
    }

    temp = ""
    for i in s:
        temp += i
        if temp.isdigit():
          answer += temp
          temp = ""
        elif temp in words:
          answer += words[temp]
          temp = ""

    return int(answer)

def solution(s):
    answer = s
    words = {
        "zero" : "0",
        "one": "1",
        "two": "2",
        "three": "3",
        "four": "4",
        "five": "5",
        "six": "6",
        "seven": "7",
        "eight": "8",
        "nine": "9",
    }

    for k, v in words.items():
        answer = answer.replace(k, v)

    return int(answer)
